#!/usr/bin/env bash
set -euo pipefail

# ====== 可改配置 ======
WATCH_DIR="${WATCH_DIR:-./}"                 # 默认监控当前目录（可用环境变量覆盖）
INTERVAL_SECONDS="${INTERVAL_SECONDS:-3600}" # 1小时=3600秒
STATE_FILE="${STATE_FILE:-./.watch.state}"   # 状态文件放在脚本同目录
LOG_FILE="${LOG_FILE:-./watch.log}"          # 日志文件放在脚本同目录

# 飞书：建议从环境变量或同目录 .env 读取
FEISHU_TOKEN="${FEISHU_TOKEN:-t-g10422cQ5SSG6NZJ3ZFIEZDYSNHJJ23DFIMHZQC2}"
FEISHU_CHAT_ID="${FEISHU_CHAT_ID:-oc_8519d60c1465a88c7c376e899718a212}"
FEISHU_API="https://open.feishu.cn/open-apis/im/v1/messages?receive_id_type=chat_id"

MSG_PREFIX="${MSG_PREFIX:-访客Dohint目录变动检测}"
# ======================

timestamp() { date '+%F %T'; }

log() {
  echo "[$(timestamp)] $*" | tee -a "$LOG_FILE" >/dev/null
}

die() {
  log "ERROR: $*"
  exit 1
}

command -v curl >/dev/null 2>&1 || die "未找到 curl，请先安装：sudo apt-get update && sudo apt-get install -y curl"
command -v find >/dev/null 2>&1 || die "未找到 find（通常系统自带）"

# 生成目录指纹：文件路径 + mtime + size（能覆盖新增/删除/修改/改名）
gen_fingerprint() {
  LC_ALL=C find "$WATCH_DIR" -type f -printf '%p|%T@|%s\n' 2>/dev/null | sort | sha256sum | awk '{print $1}'
}

send_feishu() {
  local text="$1"

  if [[ -z "$FEISHU_TOKEN" ]]; then
    log "未设置 FEISHU_TOKEN，跳过飞书发送。内容：$text"
    return 0
  fi

  # content 字段是 JSON 字符串，需要转义。这里简单处理：把双引号替换成单引号，避免破坏JSON。
  local safe_text="${text//\"/\'}"

  local payload
  payload=$(cat <<EOF
{
  "content": "{\"text\":\"$safe_text\"}",
  "msg_type": "text",
  "receive_id": "$FEISHU_CHAT_ID"
}
EOF
)

  local http_code
  http_code=$(curl -sS -o ./feishu_resp.json -w "%{http_code}" -X POST "$FEISHU_API" \
    -H "Content-Type: application/json" \
    -H "Authorization: Bearer $FEISHU_TOKEN" \
    -d "$payload")

  if [[ "$http_code" != "200" ]]; then
    log "飞书发送失败 HTTP=$http_code，响应：$(cat ./feishu_resp.json 2>/dev/null || true)"
    return 1
  fi

  log "飞书消息已发送：$safe_text"
}

init_state_if_needed() {
  [[ -d "$WATCH_DIR" ]] || die "监控目录不存在：$WATCH_DIR"

  if [[ ! -f "$STATE_FILE" ]]; then
    gen_fingerprint > "$STATE_FILE"
    log "首次运行：写入初始状态，不发送通知。目录：$WATCH_DIR"
  fi
}

check_once() {
  local current previous
  current="$(gen_fingerprint)"
  previous="$(cat "$STATE_FILE" 2>/dev/null || true)"

  if [[ "$current" != "$previous" ]]; then
    echo "$current" > "$STATE_FILE"
    local msg
    msg="$MSG_PREFIX：检测到目录发生变动
目录：$WATCH_DIR
时间：$(timestamp)"
    log "检测到变动。"
    send_feishu "$msg"
  else
    log "无变动。"
  fi
}

main_loop() {
  init_state_if_needed
  log "开始监控：$WATCH_DIR；间隔：$INTERVAL_SECONDS 秒"

  while true; do
    check_once || true
    sleep "$INTERVAL_SECONDS"
  done
}

# 如果同目录存在 .env，则自动加载（方便放 token/配置）
if [[ -f "./.env" ]]; then
  # shellcheck disable=SC1091
  source "./.env"
fi

main_loop
